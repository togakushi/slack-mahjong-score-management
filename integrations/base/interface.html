175    def get_score(self, keyword: str) -> dict:
176        """textからスコアを抽出する
177
178        Args:
179            keyword (str): 成績報告キーワード
180
181        Returns:
182            dict: 結果
183        """
184
185        text = self.data.text
186        ret: dict = {}
187
188        # 記号を置換
189        replace_chr = [
190            (chr(0xff0b), "+"),  # 全角プラス符号
191            (chr(0x2212), "-"),  # 全角マイナス符号
192            (chr(0xff08), "("),  # 全角丸括弧
193            (chr(0xff09), ")"),  # 全角丸括弧
194            (chr(0x2017), "_"),  # DOUBLE LOW LINE(半角)
195        ]
196        for z, h in replace_chr:
197            text = text.replace(z, h)
198
199        text = "".join(text.split())  # 改行削除
200
201        # パターンマッチング
202        pattern1 = re.compile(
203            rf"^({keyword})" + r"([^0-9()+-]+)([0-9+-]+)" * 4 + r"$"
204        )
205        pattern2 = re.compile(
206            r"^" + r"([^0-9()+-]+)([0-9+-]+)" * 4 + rf"({keyword})$"
207        )
208        pattern3 = re.compile(
209            rf"^({keyword})\((.+?)\)" + r"([^0-9()+-]+)([0-9+-]+)" * 4 + r"$"
210        )
211        pattern4 = re.compile(
212            r"^" + r"([^0-9()+-]+)([0-9+-]+)" * 4 + rf"({keyword})\((.+?)\)$"
213        )
214
215        # 情報取り出し
216        position: dict[str, int] = {}
217        match text:
218            case text if pattern1.findall(text):
219                msg = pattern1.findall(text)[0]
220                position = {
221                    "p1_name": 1, "p1_str": 2,
222                    "p2_name": 3, "p2_str": 4,
223                    "p3_name": 5, "p3_str": 6,
224                    "p4_name": 7, "p4_str": 8,
225                }
226                comment = None
227            case text if pattern2.findall(text):
228                msg = pattern2.findall(text)[0]
229                position = {
230                    "p1_name": 0, "p1_str": 1,
231                    "p2_name": 2, "p2_str": 3,
232                    "p3_name": 4, "p3_str": 5,
233                    "p4_name": 6, "p4_str": 7,
234                }
235                comment = None
236            case text if pattern3.findall(text):
237                msg = pattern3.findall(text)[0]
238                position = {
239                    "p1_name": 2, "p1_str": 3,
240                    "p2_name": 4, "p2_str": 5,
241                    "p3_name": 6, "p3_str": 7,
242                    "p4_name": 8, "p4_str": 9,
243                }
244                comment = str(msg[1])
245            case text if pattern4.findall(text):
246                msg = pattern4.findall(text)[0]
247                position = {
248                    "p1_name": 0, "p1_str": 1,
249                    "p2_name": 2, "p2_str": 3,
250                    "p3_name": 4, "p3_str": 5,
251                    "p4_name": 6, "p4_str": 7,
252                }
253                comment = str(msg[9])
254            case _:
255                return ret
256
257        for k, p in position.items():
258            ret.update({k: str(msg[p])})
259
260        ret.update(comment=comment)
261        ret.update(ts=self.data.event_ts)
262
263        return ret
264
265    def get_remarks(self, keyword: str) -> list:
266        """textからメモを抽出する
267
268        Args:
269            keyword (str): メモ記録キーワード
270
271        Returns:
272            list: 結果
273        """
274
275        ret: list = []
276        if re.match(rf"^{keyword}", self.data.text):  # キーワードが先頭に存在するかチェック
277            text = self.data.text.replace(keyword, "").strip().split()
278            for name, matter in zip(text[0::2], text[1::2]):
279                ret.append([name, matter])
280
281        return ret
