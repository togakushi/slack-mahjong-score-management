 35def get_game_results() -> list:
 36    """月/年単位のゲーム結果集計
 37
 38    Returns:
 39        list: 集計結果のリスト

 63            "",

 65            "",
 66            "4位",
 67            "",
 68            "平均\n順位",
 69            "トビ",
 70            "",
 71        ]
 72    ]
 73
 74    for row in rows.fetchall():
 75        if row["ゲーム数"] == 0:

376def gen_pdf(m: "MessageParserProtocol"):
377    """成績レポートを生成する
378
379    Args:
380        m (MessageParserProtocol): メッセージデータ
381    """
382
383    if g.adapter.conf.plotting_backend == "plotly":
384        m.post.reset()
385        m.post.headline = {"": message.random_reply(m, "not_implemented")}
386        return
387
388    if not g.params.get("player_name"):  # レポート対象の指定なし
389        m.post.headline = {"成績レポート": message.random_reply(m, "no_target")}
390        m.status.result = False
391        return
392
393    # 対象メンバーの記録状況
394    target_info = lookup.member_info(g.params)
395    logging.debug(target_info)
396
397    if not target_info["game_count"]:  # 記録なし
398        m.post.headline = {"成績レポート": message.random_reply(m, "no_hits")}
399        m.status.result = False
400        return
401
402    # 書式設定
403    font_path = os.path.join(os.path.realpath(os.path.curdir), g.cfg.setting.font_file)
404    pdf_path = g.cfg.setting.work_dir / (f"{g.params['filename']}.pdf" if g.params.get("filename") else "results.pdf")
405    pdfmetrics.registerFont(TTFont("ReportFont", font_path))
406
407    doc = SimpleDocTemplate(
408        str(pdf_path),
409        pagesize=landscape(A4),
410        topMargin=10.0 * mm,
411        bottomMargin=10.0 * mm,
412        # leftMargin=1.5 * mm,
413        # rightMargin=1.5 * mm,
414    )
415
416    style: dict = {}
417    style["Title"] = ParagraphStyle(name="Title", fontName="ReportFont", fontSize=24)
418    style["Normal"] = ParagraphStyle(name="Normal", fontName="ReportFont", fontSize=14)
419    style["Left"] = ParagraphStyle(name="Left", fontName="ReportFont", fontSize=14, alignment=TA_LEFT)
420    style["Right"] = ParagraphStyle(name="Right", fontName="ReportFont", fontSize=14, alignment=TA_RIGHT)
421
422    plt.rcParams.update(plt.rcParamsDefault)
423    font_prop = fm.FontProperties(fname=font_path)
424    plt.rcParams["font.family"] = font_prop.get_name()
425    fm.fontManager.addfont(font_path)
426
427    # レポート作成
428    elements: list = []
429    elements.extend(cover_page(style, target_info))  # 表紙
430    elements.extend(entire_aggregate(style))  # 全期間
431    elements.extend(periodic_aggregation(style))  # 期間集計
432    elements.extend(sectional_aggregate(style, target_info))  # 区間集計
433
434    doc.build(elements)
435    logging.debug("report generation: %s", g.params["player_name"])
436
437    m.set_data(pdf_path, StyleOptions(title=f"成績レポート({g.params['player_name']})", use_comment=True, header_hidden=True))

440def cover_page(style: dict, target_info: dict) -> list:
441    """表紙生成
442
443    Args:
444        style (dict): レイアウトスタイル
445        target_info (dict): プレイヤー情報
446
447    Returns:
448        list: 生成内容
449    """
450
451    elements: list = []
452
453    first_game = datetime.fromtimestamp(  # 最初のゲーム日時
454        float(target_info["first_game"])
455    )
456    last_game = datetime.fromtimestamp(  # 最後のゲーム日時
457        float(target_info["last_game"])
458    )
459
460    if g.params.get("anonymous"):
461        mapping_dict = formatter.anonymous_mapping([g.params["player_name"]])
462        target_player = next(iter(mapping_dict.values()))
463    else:
464        target_player = g.params["player_name"]
465
466    # 表紙
467    elements.append(Spacer(1, 40 * mm))
468    elements.append(Paragraph(f"成績レポート：{target_player}", style["Title"]))
469    elements.append(Spacer(1, 10 * mm))
470    elements.append(
471        Paragraph(
472            f"集計期間：{first_game.strftime('%Y-%m-%d %H:%M')} - {last_game.strftime('%Y-%m-%d %H:%M')}",
473            style["Normal"],
474        )
475    )
476    elements.append(Spacer(1, 100 * mm))
477    elements.append(Paragraph(f"作成日：{datetime.now().strftime('%Y-%m-%d')}", style["Right"]))
478    elements.append(PageBreak())
479
480    return elements

580def periodic_aggregation(style: dict) -> list:
581    """期間集計
582
583    Args:
584        style (dict): レイアウトスタイル
585
586    Returns:
587        list: 生成内容
588    """
589
590    elements: list = []
591
592    pattern: list[tuple[str, str, Literal["A", "M", "Y"]]] = [
593        # 表タイトル, グラフタイトル, フラグ
594        ("月別集計", "順位分布（月別）", "M"),
595        ("年別集計", "順位分布（年別）", "Y"),
596    ]
597
598    for table_title, graph_title, flag in pattern:
599        elements.append(Paragraph(table_title, style["Left"]))
600        elements.append(Spacer(1, 5 * mm))
601
602        data: list = []
603        g.cfg.aggregate_unit = flag
604        tmp_data = get_game_results()
605
606        if not tmp_data:
607            return []
608
609        for _, val in enumerate(tmp_data):  # 日時を除外
610            data.append(val[:15])
611
612        tt = LongTable(data, repeatRows=1)
613        ts = TableStyle(
614            [
615                ("FONT", (0, 0), (-1, -1), "ReportFont", 10),
616                ("GRID", (0, 0), (-1, -1), 0.25, colors.black),
617                ("VALIGN", (0, 0), (-1, -1), "MIDDLE"),
618                ("ALIGN", (0, 0), (-1, -1), "CENTER"),
619                ("SPAN", (4, 0), (5, 0)),
620                ("SPAN", (6, 0), (7, 0)),
621                ("SPAN", (8, 0), (9, 0)),
622                ("SPAN", (10, 0), (11, 0)),
623                ("SPAN", (13, 0), (14, 0)),
624                # ヘッダ行
625                ("BACKGROUND", (0, 0), (-1, 0), colors.navy),
626                ("TEXTCOLOR", (0, 0), (-1, 0), colors.white),
627            ]
628        )
629
630        if len(data) > 4:
631            for i in range(len(data) - 2):
632                if i % 2 == 0:
633                    ts.add("BACKGROUND", (0, i + 2), (-1, i + 2), colors.lightgrey)
634        tt.setStyle(ts)
635        elements.append(tt)
636        elements.append(Spacer(1, 10 * mm))
637
638        # 順位分布
639        df = pd.DataFrame(
640            {
641                "1位率": [float(str(data[x + 1][5]).replace("%", "")) for x in range(len(data) - 1)],
642                "2位率": [float(str(data[x + 1][7]).replace("%", "")) for x in range(len(data) - 1)],
643                "3位率": [float(str(data[x + 1][9]).replace("%", "")) for x in range(len(data) - 1)],
644                "4位率": [float(str(data[x + 1][11]).replace("%", "")) for x in range(len(data) - 1)],
645            },
646            index=[data[x + 1][0] for x in range(len(data) - 1)],
647        )
648
649        imgdata = graphing_rank_distribution(df, graph_title)
650        elements.append(Spacer(1, 5 * mm))
651        elements.append(Image(imgdata, width=1200 * 0.5, height=700 * 0.5))
652
653        elements.append(PageBreak())
654
655    return elements
