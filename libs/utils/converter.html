 20def save_output(
 21    df: pd.DataFrame,
 22    kind: str,
 23    filename: str,
 24    index: bool = False,
 25    headline: Optional[str] = None,
 26    suffix: Optional[str] = None,
 27) -> Union["Path", None]:
 28    """指定されたフォーマットでdfを保存する
 29
 30    Args:
 31        df (pd.DataFrame): 描写対象データ
 32        kind (str): フォーマット
 33        filename (str): 保存ファイル名
 34        index (bool): インデックスを含める. Defaults to False.
 35        headline (Optional[str], optional): 集計情報（ヘッダコメント）. Defaults to None.
 36        suffix (Optional[str], optional): 保存ファイル名に追加する文字列. Defaults to None.
 37
 38    Returns:
 39        Path: 保存したファイルパス
 40        None: 未知のフォーマットが指定された場合
 41    """
 42
 43    match kind.lower():
 44        case "csv":
 45            data = df.to_csv(index=index)
 46        case "text" | "txt":
 47            data = df.to_markdown(
 48                index=index,
 49                tablefmt="outline",
 50                floatfmt=formatter.floatfmt_adjust(df, index=index),
 51                colalign=formatter.column_alignment(df, index=index),
 52                # headersalign=column_alignment(df, True),  # ToDo: python-tabulate >= 0.10.0
 53            )
 54        case _:
 55            return None
 56
 57    # 保存
 58    save_file = textutil.save_file_path(filename, True)
 59    if suffix and g.params.get("filename"):
 60        save_file = save_file.with_name(f"{save_file.stem}_{suffix}{save_file.suffix}")
 61
 62    with open(save_file, "w", encoding="utf-8") as writefile:
 63        if headline is not None:  # ヘッダ書き込み
 64            for line in headline.splitlines():
 65                writefile.writelines(f"# {line}\n")
 66            writefile.writelines("\n")
 67
 68        writefile.writelines(data)
 69
 70    return save_file

 73def df_to_text_table(df: pd.DataFrame, step: int = 40, index: bool = False) -> dict:
 74    """DataFrameからテキストテーブルの生成
 75
 76    Args:
 77        df (pd.DataFrame): 対象データ
 78        step (int, optional): 分割行. Defaults to 40.
 79        index (bool, optional): インデックスを含める. Defaults to False.
 80
 81    Returns:
 82        dict: 生成テーブル
 83    """
 84
 85    # ヘッダ/位置
 86    header: list = []
 87    alignments: list = []
 88    if index:
 89        df.reset_index(inplace=True, drop=True)
 90        df.index += 1
 91        header.append("")
 92    for col in df.columns:
 93        header.append(col)
 94        match col:
 95            case "名前" | "プレイヤー名":
 96                alignments.append(Alignment.LEFT)
 97            case "順位分布":
 98                alignments.append(Alignment.LEFT)
 99            case _:
100                alignments.append(Alignment.RIGHT)
101
102    # 表データ
103    body: list = []
104    data: list = []
105    for row in df.to_dict(orient="records"):
106        data.clear()
107        for k, v in row.items():
108            match k:
109                case "通算" | "平均" | "平均素点":
110                    data.append(f" {v:+.1f}".replace(" -", "▲"))
111                case "平順" | "平均順位":
112                    data.append(f"{v:.2f}")
113                case "レート":
114                    data.append(f"{v:.1f}")
115                case "順位偏差" | "得点偏差":
116                    data.append(f"{v:.0f}")
117                case _:
118                    data.append(str(v).replace("nan", "*****"))
119            if index:
120                data.insert(0, "")
121        body.append(data.copy())
122
123    # 表生成/分割
124    my_style = PresetStyle.plain
125    my_style.heading_row_sep = "-"
126    my_style.heading_row_right_tee = ""
127    my_style.heading_row_left_tee = ""
128
129    table_data: dict = {}
130    for idx, table_body in enumerate(textutil.split_balanced(body, step)):
131        output = table2ascii(
132            header=header,
133            body=table_body,
134            style=my_style,
135            cell_padding=0,
136            first_col_heading=index,
137            alignments=alignments,
138        )
139        table_data.update({f"{idx}": output})
140
141    return table_data

144def df_to_text_table2(df: pd.DataFrame, style: "StyleOptions", limit: int = 2000) -> dict:
145    """DataFrameからテキストテーブルの生成(縦横変換)
146
147    Args:
148        df (pd.DataFrame): 対象データ
149        style (StyleOptions): 表示フラグ
150        limit (int, optional): 分割文字数. Defaults to 2000.
151
152    Returns:
153        dict: 生成テーブル
154    """
155
156    # 表生成/分割
157    my_style = PresetStyle.plain
158    my_style.heading_row_sep = "-"
159    my_style.heading_row_right_tee = ""
160    my_style.heading_row_left_tee = ""
161    my_style.heading_col_sep = "： "
162
163    table_data: dict = {}
164    start_block: int = 0
165
166    safe_output: str = ""
167
168    for cur_block in range(len(df.columns)):
169        chk_df = df.iloc[:, start_block:cur_block]
170
171        # ヘッダ
172        header: list = chk_df.columns.to_list()
173        if style.show_index:
174            header.insert(0, "")
175
176        # ボディ
177        body: list = []
178        data: list = []
179        for idx, item in chk_df.iterrows():
180            data.clear()
181            data.append(idx)
182            data.extend(item.to_list())
183            body.append(data.copy())
184
185        output = table2ascii(
186            header=header,
187            body=body,
188            style=my_style,
189            cell_padding=0,
190            alignments=[Alignment.RIGHT] * len(header),
191            first_col_heading=style.show_index,
192        )
193
194        # 文字数チェック
195        if len(output) < limit:
196            safe_output = output
197        else:
198            table_data.update({f"{cur_block}": safe_output})
199            start_block = cur_block - 1
200    # 最終ブロック
201    if cur_block != len(df.columns):
202        table_data.update({f"{cur_block}": output})
203
204    return table_data

207def df_to_results_details(df: pd.DataFrame) -> dict:
208    """戦績(詳細)データをテキスト変換
209
210    Args:
211        df (pd.DataFrame): 対象データ
212
213    Returns:
214        dict: 整形テキスト
215    """
216
217    data_list: list = []
218    for x in df.to_dict(orient="index").values():
219        work_df = pd.DataFrame({
220            "東家：": [v for k, v in cast(dict, x).items() if str(k).startswith("東家")],
221            "南家：": [v for k, v in cast(dict, x).items() if str(k).startswith("南家")],
222            "西家：": [v for k, v in cast(dict, x).items() if str(k).startswith("西家")],
223            "北家：": [v for k, v in cast(dict, x).items() if str(k).startswith("北家")],
224        }, index=["name", "rpoint", "rank", "point", "grandslam"]).T
225
226        work_df["rpoint"] = work_df.apply(lambda v: f"<>{v["rpoint"]:8d}点".replace("-", "▲"), axis=1)
227        work_df["point"] = work_df.apply(lambda v: f"(<>{v["point"]:7.1f}pt)".replace("-", "▲"), axis=1)
228        work_df["rank"] = work_df.apply(lambda v: f"{v["rank"]}位", axis=1)
229        data = work_df.to_markdown(tablefmt="tsv", headers=[], floatfmt=formatter.floatfmt_adjust(work_df)).replace("<>", "")
230
231        ret = f"{str(x["日時"]).replace("-", "/")} {x["備考"]}\n"
232        ret += textwrap.indent(data, "\t") + "\n"
233
234        data_list.append(ret)
235
236    return {str(idx): x for idx, x in enumerate(formatter.group_strings(data_list, 2000))}

441def df_to_remarks(df: pd.DataFrame) -> dict:
442    """DataFrameからメモテーブルを生成
443
444    Args:
445        df (pd.DataFrame): 対象データ
446
447    Returns:
448        dict: 整形テキスト
449    """
450
451    for col in df.columns:
452        match col:
453            case "日時":
454                df["日時"] = df["日時"].map(lambda x: str(x).replace("-", "/"))
455            case "卓外":
456                df["卓外"] = df["卓外"].map(lambda x: f"{x}pt".replace("-", "▲"))
457
458    if "卓外" in df.columns:
459        df["表示"] = df.apply(lambda x: f"{x["日時"]} {x["内容"]} {x["卓外"]} ({x["名前"]})", axis=1)
460    elif "和了役" in df.columns:
461        df["表示"] = df.apply(lambda x: f"{x["日時"]} {x["和了役"]} ({x["名前"]})", axis=1)
462    else:
463        df["表示"] = df.apply(lambda x: f"{x["日時"]} {x["内容"]} ({x["名前"]})", axis=1)
464
465    tbl = tabulate(df.filter(items=["表示"]).values, showindex=False).splitlines()[1:-1]
466
467    return {"0": "\n".join(tbl)}
