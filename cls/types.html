121    def initialization(self, section: str | None = None) -> None:
122        """設定ファイルから値を取りこみ"""
123        config = getattr(self, "config")
124        assert config is not None, "config must not be None"
125
126        if section is None:
127            section = getattr(self, "section")
128        assert section is not None, "section must not be None"
129
130        for x in fields(self):
131            if x.type == Union[ConfigParser | None]:
132                continue
133            if x.type == Union[str | None] and x.name == "section":
134                setattr(self, x.name, section)
135            elif x.type == bool:
136                setattr(self, x.name, config.getboolean(section, x.name, fallback=x.default))
137            elif x.type == str:
138                setattr(self, x.name, config.get(section, x.name, fallback=x.default))
139            elif x.type == int:
140                setattr(self, x.name, config.getint(section, x.name, fallback=x.default))
141            elif x.type == float:
142                setattr(self, x.name, config.getfloat(section, x.name, fallback=x.default))
143            elif x.type == list:
144                tmp_list: list = []
145                for data in config.get(section, x.name, fallback="").split(","):
146                    tmp_list.extend(data.split())
147                if x.name == "delete":
148                    for data in config.get(section, "del", fallback="").split(","):
149                        tmp_list.extend(data.split())
150                setattr(self, x.name, tmp_list)
151            else:
152                setattr(self, x.name, config.get(section, x.name, fallback=x.default))
153
154        # 共通パラメータ初期化
155        self.format = str()
156        self.filename = str()
157        self.aggregate_unit = str()
158        self.interval = 80
159
160    def to_dict(self) -> dict:
161        """必要なパラメータを辞書型で返す
162
163        Returns:
164            dict: 返却値
165        """
166
167        ret_dict: dict = asdict(self)
168        ret_dict.update(format=getattr(self, "format", ""))
169        ret_dict.update(filename=getattr(self, "filename", ""))
170        ret_dict.update(interval=getattr(self, "interval", 80))
171
172        drop_keys: list = [
173            "config",
174            "section",
175            "always_argument",
176            "regulations_type2",
177            "rank_point",
178        ]
179
180        for key in drop_keys:
181            if key in ret_dict:
182                ret_dict.pop(key)
183
184        return ret_dict
185
186    def get_default(self, attr: str) -> Any:
187        """デフォルト値を取得して返す
188
189        Args:
190            attr (str): 属性
191
192        Raises:
193            AttributeError: 未定義
194
195        Returns:
196            Any: デフォルト値
197        """
198
199        ret: Any
200
201        for x in fields(self):
202            if x.name == attr:
203                config = getattr(self, "config")
204                assert config is not None, "config must not be None"
205                section = getattr(self, "section")
206                assert section is not None, "section must not be None"
207
208                if x.type == Union[str | None]:
209                    ret = None
210                elif x.type == bool:
211                    ret = config.getboolean(section, x.name, fallback=x.default)
212                elif x.type == str:
213                    ret = config.get(section, x.name, fallback=x.default)
214                elif x.type == int:
215                    ret = config.getint(section, x.name, fallback=x.default)
216                elif x.type == float:
217                    ret = config.getfloat(section, x.name, fallback=x.default)
218                elif x.type == list:
219                    ret = []
220                else:
221                    ret = config.get(section, x.name, fallback=x.default)
222                return ret
223
224        raise AttributeError(f"{attr} has no default or does not exist.")
