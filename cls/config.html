 94        """"""
 95        for k in self._section.keys():
 96            match type(self.__dict__.get(k)):
 97                case v_type if k in self.__dict__ and v_type is str:
 98                    setattr(self, k, self._section.get(k, fallback=self.get(k)))
 99                case v_type if k in self.__dict__ and v_type is int:
100                    setattr(self, k, self._section.getint(k, fallback=self.get(k)))
101                case v_type if k in self.__dict__ and v_type is float:

103                case v_type if v_type is bool:

107                    current_list = getattr(self, k)
108                    if isinstance(current_list, list) and current_list:  #

110                    else:
111                        setattr(self, k, v_list)
112                case v_type if k in self.__dict__ and v_type is Optional[str]:  #  or None()
113                    setattr(self, k, self._section.get(k, fallback=self.get(k)))
114                case v_type if k in self.__dict__ and v_type is PosixPath:
