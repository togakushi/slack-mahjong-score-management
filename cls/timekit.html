cls.timekit

timekit - datetime 拡張ユーティリティ

  • ExtendedDatetime: 柔軟な初期化と書式変換ができる datetime 拡張クラス
  • ExtendedDatetimeList: ExtendedDatetimeを要素とする日付リストを扱う補助クラス
Examples:
>>> from cls.timekit import ExtendedDatetime
>>> t = ExtendedDatetime("2025-04-19 12:34:56")
>>> t.format("ymdhm")
'2025/04/19 12:34'

>>> t.set("2025-05-01 00:00:00")
>>> t.format("sql")
'2025-05-01 00:00:00.000000'

>>> from dateutil.relativedelta import relativedelta
>>> t2 = t + relativedelta(days=93)
>>> t2.format("ymd")
'2025/08/02'

>>> t + {"days": 1, "months": 2}
2025-07-02 00:00:00.000000

>>> ExtendedDatetime.range("今月").format("ymdhm")
['2025/04/01 00:00', '2025/04/30 23:59']

>>> ExtendedDatetime.range("今月").dict_format("ymd", "ja")
{'start': '2025年04月01日', 'end': '2025年04月30日'}

>>> ExtendedDatetime("2025-01-01 01:23:45", hours=-12).range("今年")
[2024-01-01 00:00:00.000000, 2024-12-31 23:59:59.999999]

154@total_ordering
155class ExtendedDatetime:
156    """datetime拡張クラス"""
157
158    _dt: datetime
159    """操作対象"""
160
161    # 型アノテーション用定数
162    AcceptedType: TypeAlias = Union[str, float, datetime, "ExtendedDatetime"]
163    """引数として受け付ける型
164    - **str**: 日付文字列（ISO形式など）
165    - **float**: UNIXタイムスタンプ
166    - **datetime** / **ExtendedDatetime**: オブジェクトをそのまま利用
167    """
168
169    FormatType: TypeAlias = FormatType
170    DelimiterStyle: TypeAlias = DelimiterStyle
171
172    def __init__(self, value: AcceptedType | None = None, **relativedelta_kwargs):
173        """ExtendedDatetimeの初期化
174
175        Args:
176            value (AcceptedType | None, optional): 引数
177                - None: 現在時刻(`datetime.now()`)で初期化
178            relativedelta_kwargs (dict): 初期化時にrelativedelta()に渡す引数
179        """
180
181        self._dt = self.convert(value) if value else datetime.now()
182        if relativedelta_kwargs:
183            self._dt += relativedelta(**relativedelta_kwargs)
184
185    def __str__(self) -> str:
186        return self.format("sql")
187
188    def __repr__(self) -> str:
189        return self.format("sql")
190
191    def __eq__(self, other):
192        if isinstance(other, ExtendedDatetime):
193            return self.dt == other.dt
194        if isinstance(other, datetime):
195            return self.dt == other
196        return NotImplemented
197
198    def __lt__(self, other):
199        if isinstance(other, ExtendedDatetime):
200            return self.dt < other.dt
201        if isinstance(other, datetime):
202            return self.dt < other
203        return NotImplemented
204
205    def __add__(self, other: Union[relativedelta, dict]) -> "ExtendedDatetime":
206        if isinstance(other, dict):
207            delta = relativedelta(**other)
208        elif isinstance(other, relativedelta):
209            delta = other
210        else:
211            raise TypeError("Expected dict or relativedelta")
212
213        return ExtendedDatetime(self._dt + delta)
214
215    def __sub__(self, other: Union[relativedelta, dict]) -> "ExtendedDatetime":
216        if isinstance(other, dict):
217            delta = relativedelta(**other)
218        elif isinstance(other, relativedelta):
219            delta = other
220        else:
221            raise TypeError("Expected dict or relativedelta")
222
223        return ExtendedDatetime(self._dt - delta)
224
225    def __radd__(self, other: Union[relativedelta, dict]) -> "ExtendedDatetime":
226        return self.__add__(other)
227
228    def __rsub__(self, other: Union[relativedelta, dict]) -> "ExtendedDatetime":
229        return self.__sub__(other)
230
231    def __hash__(self):
232        return hash(self.dt)
233
234    def __getattr__(self, name):
235        return getattr(self._dt, name)

256    def format(self, fmt: FormatType, delimiter: DelimiterStyle = None) -> str:
257        """フォーマット変換
258
259        Args:
260            fmt (FormatType): 変換形式
261            delimiter (DelimiterStyle): 区切り
262
263        Raises:
264            ValueError: 受け付けない変換形式
265
266        Returns:
267            str: 変換文字列
268        """
269
270        ret: str
271        match fmt:
272            case "ts":
273                ret = str(self._dt.timestamp())
274            case "y":
275                match delimiter:
276                    case "ja":
277                        ret = self._dt.strftime("%Y年")
278                    case _:
279                        ret = self._dt.strftime("%Y")
280            case "ym":
281                match delimiter:
282                    case "slash" | "/":
283                        ret = self._dt.strftime("%Y/%m")
284                    case "hyphen" | "-":
285                        ret = self._dt.strftime("%Y-%m")
286                    case "ja":
287                        ret = self._dt.strftime("%Y年%m月")
288                    case "number" | "num":
289                        ret = self._dt.strftime("%Y%m")
290                    case _:
291                        ret = self._dt.strftime("%Y/%m")
292            case "ymd":
293                match delimiter:
294                    case "slash" | "/":
295                        ret = self._dt.strftime("%Y/%m/%d")
296                    case "hyphen" | "-":
297                        ret = self._dt.strftime("%Y-%m-%d")
298                    case "ja":
299                        ret = self._dt.strftime("%Y年%m月%d日")
300                    case "number" | "num":
301                        ret = self._dt.strftime("%Y%m%d")
302                    case _:
303                        ret = self._dt.strftime("%Y/%m/%d")
304            case "ymdhm":
305                match delimiter:
306                    case "slash" | "/":
307                        ret = self._dt.strftime("%Y/%m/%d %H:%M")
308                    case "hyphen" | "-":
309                        ret = self._dt.strftime("%Y-%m-%d %H:%M")
310                    case "ja":
311                        ret = self._dt.strftime("%Y年%m月%d日 %H時%M分")
312                    case "number" | "num":
313                        ret = self._dt.strftime("%Y%m%d%H%M")
314                    case _:
315                        ret = self._dt.strftime("%Y/%m/%d %H:%M")
316            case "ymdhms":
317                match delimiter:
318                    case "slash" | "/":
319                        ret = self._dt.strftime("%Y/%m/%d %H:%M:%S")
320                    case "hyphen" | "-":
321                        ret = self._dt.strftime("%Y-%m-%d %H:%M:%S")
322                    case "ja":
323                        ret = self._dt.strftime("%Y年%m月%d日 %H時%M分%S秒")
324                    case "number" | "num":
325                        ret = self._dt.strftime("%Y%m%d%H%M%S")
326                    case _:
327                        ret = self._dt.strftime("%Y/%m/%d %H:%M:%S")
328            case "sql":
329                match delimiter:
330                    case "slash" | "/":
331                        ret = self._dt.strftime("%Y/%m/%d %H:%M:%S.%f")
332                    case "hyphen" | "-":
333                        ret = self._dt.strftime("%Y-%m-%d %H:%M:%S.%f")
334                    case "number" | "num":
335                        ret = self._dt.strftime("%Y%m%d%H%M%S%f")
336                    case _:
337                        ret = self._dt.strftime("%Y-%m-%d %H:%M:%S.%f")
338            case "ext":
339                ret = self._dt.strftime("%Y%m%d-%H%M%S")
340            case _:
341                raise ValueError(f"Unknown format: {fmt}")
342
343        return ret

345    def range(self, value: str | list) -> "ExtendedDatetimeList":
346        """キーワードが示す範囲をリストで返す
347
348        Args:
349            value (str | list): 範囲取得キーワード
350                - str: スペース区切りで分割してリスト化
351                - list: スペース区切りで再分割
352
353        Returns:
354            ExtendedDatetimeList: 日付リスト
355        """
356
357        if isinstance(value, str):
358            check_list = value.split()
359        else:
360            check_list = sum([str(x).split() for x in value], [])  # 平坦化
361
362        ret: list[datetime] = []
363        for word in check_list:
364            for _, range_map in DATE_RANGE_MAP.items():
365                if word in cast(list, range_map["keyword"]):
366                    ret.extend(range_map["range"](self._dt))
367                    break
368            else:
369                try:
370                    try_time = self.convert(str(word))
371                    ret.append(try_time.replace(hour=0, minute=0, second=0, microsecond=0))
372                    ret.append(try_time.replace(hour=23, minute=59, second=59, microsecond=999999))
373                except ValueError:
374                    pass
375
376            continue
377
378        return ExtendedDatetimeList([ExtendedDatetime(x) for x in ret])

381    def valid_keywords(cls) -> list[str]:
382        """有効なキーワード一覧
383

394    @classmethod
395    def print_range(cls) -> str:
396        """指定可能キーワードで取得できる範囲の一覧
397
398        Returns:
399            str: 出力メッセージ
400        """
401
402        base_instance = cls()
403        ret: str = ""
404
405        for _, val in DATE_RANGE_MAP.items():
406            for label in val["keyword"]:
407                scope = " ～ ".join(base_instance.range(label).format("ymd"))
408                ret += f"{label}：{scope}\n"
409
410        return ret.strip()

412    @staticmethod
413    def convert(value: AcceptedType) -> datetime:
414        """引数の型を判定してdatetimeへ変換
415
416        Args:
417            value (AcceptedType): 変換対象
418
419        Raises:
420            TypeError: str型が変換できない場合
421
422        Returns:
423            datetime: 変換した型
424        """
425
426        if isinstance(value, ExtendedDatetime):
427            return value.dt
428        if isinstance(value, datetime):
429            return value
430        if isinstance(value, float):
431            return datetime.fromtimestamp(value)
432        if isinstance(value, str):
433            try:
434                return datetime.fromisoformat(value)
435            except ValueError:
436                return datetime.strptime(value, "%Y/%m/%d %H:%M")
437
438        raise TypeError("Unsupported type for datetime conversion")

441class ExtendedDatetimeList(list):
442    """ExtendedDatetimeを要素とする日付リストを扱う補助クラス"""
443
444    FormatType: TypeAlias = FormatType
445    DelimiterStyle: TypeAlias = DelimiterStyle
446
447    def __add__(self, other):
448        if isinstance(other, dict):
449            return ExtendedDatetimeList([dt + other for dt in self])
450        return NotImplemented
451
452    def __sub__(self, other):
453        if isinstance(other, dict):
454            return ExtendedDatetimeList([dt - other for dt in self])
455        return NotImplemented
456
457    @property
458    def start(self) -> ExtendedDatetime | None:
459        """最小日付を返す。空ならNone。"""
460        return (min(self) if self else None)
461
462    @property
463    def end(self) -> ExtendedDatetime | None:
464        """最大日付を返す。空ならNone。"""
465        return (max(self) if self else None)
466
467    @property
468    def period(self) -> List[ExtendedDatetime | None]:
469        """最小値と最大値をリストで返す"""
470        min_dt = min(self) if self else None
471        max_dt = max(self) if self else None
472
473        return [min_dt, max_dt]
474
475    def format(self, fmt: FormatType = "sql", delimiter: DelimiterStyle = None) -> list[str]:
476        """全要素にformatを適用した文字列リストを返す
477
478        Args:
479            fmt (FormatType, optional): フォーマット変換. Defaults to "sql".
480            delimiter (DelimiterStyle, optional): 区切り記号指定. Defaults to None.
481
482        Returns:
483            list[str]: 生成したリスト
484        """
485
486        return [dt.format(fmt, delimiter) for dt in self if isinstance(dt, ExtendedDatetime)]
487
488    def dict_format(self, fmt: FormatType = "sql", delimiter: DelimiterStyle = None) -> dict[str, str]:
489        """全要素にformatを適用し、最小日付と最大日付を辞書で返す
490
491        Args:
492            fmt (FormatType, optional): フォーマット変換. Defaults to "sql".
493            delimiter (DelimiterStyle, optional): 区切り記号指定. Defaults to None.
494
495        Returns:
496            dict[str, str]: 生成した辞書
497        """
498
499        date_range = [dt for dt in self if isinstance(dt, ExtendedDatetime)]
500
501        if not date_range:
502            return {}
503
504        return ({"start": min(date_range).format(fmt, delimiter), "end": max(date_range).format(fmt, delimiter)})
