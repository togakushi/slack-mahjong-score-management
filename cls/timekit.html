cls.timekit

timekit - datetime 拡張ユーティリティ

ExtendedDatetime: 柔軟な初期化と書式変換ができる datetime ラッパー

Examples:
>>> from cls.timekit import ExtendedDatetime
>>> t = ExtendedDatetime("2025-04-19 12:34:56")
>>> t.format("ymdhm")
'2025/04/19 12:34'

>>> t.set("2025-05-01 00:00:00")
>>> t.format("sql")
'2025-05-01 00:00:00.000000'

>>> from dateutil.relativedelta import relativedelta
>>> t2 = t + relativedelta(days=93)
>>> t2.format("ymd")
'2025/08/02'

>>> t + {"days": 1, "months": 2}
2025-07-02 00:00:00.000000

>>> ExtendedDatetime.get_range("今月").format("ymdhm")
['2025/04/01 00:00', '2025/04/30 23:59']

>>> ExtendedDatetime.get_range("今月").dict_format("ymd", "ja")
{'start': '2025年04月01日', 'end': '2025年04月30日'}

268    @classmethod
269    def _register_keywords(cls):
270        """範囲指定キーワードから日時を取得する"""
271        current_time = datetime.now().replace(hour=0, minute=0, second=0, microsecond=0)
272        appointed_time = (datetime.now() + relativedelta(hours=-12)).replace(hour=12, minute=0, second=0, microsecond=0)
273        try:
274            game_first_record = lookup.db.first_record() + relativedelta(days=-1)
275        except AttributeError:
276            game_first_record = datetime.fromtimestamp(0)
277
278        def add_keywords(keys: list[str], func: Callable[[], list[datetime]]):
279            for key in keys:
280                cls._range_map[key] = func
281
282        add_keywords(["当日"], lambda: [appointed_time])
283        add_keywords(["今日"], lambda: [current_time])
284        add_keywords(["昨日"], lambda: [current_time + relativedelta(days=-1)])
285        add_keywords(["今月"], lambda: [
286            appointed_time.replace(day=1, hour=0),
287            appointed_time + relativedelta(day=31, hour=23, minute=59, second=59, microsecond=999999),
288        ])
289        add_keywords(["先月"], lambda: [
290            appointed_time + relativedelta(months=-1, day=1, hour=0),
291            appointed_time + relativedelta(months=-1, day=31, hour=23, minute=59, second=59, microsecond=999999),
292        ])
293        add_keywords(["先々月"], lambda: [
294            appointed_time + relativedelta(months=-2, day=1),
295            appointed_time + relativedelta(months=-2, day=31, hour=23, minute=59, second=59, microsecond=999999),
296        ])
297        add_keywords(["今年"], lambda: [
298            current_time.replace(month=1, day=1),
299            current_time.replace(month=12, day=31, hour=23, minute=59, second=59, microsecond=999999),
300        ])
301        add_keywords(["昨年", "去年"], lambda: [
302            current_time + relativedelta(years=-1, month=1, day=1),
303            current_time + relativedelta(years=-1, month=12, day=31, hour=23, minute=59, second=59, microsecond=999999),
304        ])
305        add_keywords(["一昨年"], lambda: [
306            current_time + relativedelta(years=-2, month=1, day=1),
307            current_time + relativedelta(years=-2, month=12, day=31, hour=23, minute=59, second=59, microsecond=999999),
308        ])
309        add_keywords(["最初"], lambda: [game_first_record])
310        add_keywords(["最後"], lambda: [current_time + relativedelta(days=1)])
311        add_keywords(["全部"], lambda: [
312            game_first_record,
313            current_time + relativedelta(days=1, hour=23, minute=59, second=59, microsecond=999999),
314        ])

316    def _convert(self, value: AcceptedTypes) -> datetime:
317        """引数の型を判定してdatetimeへ変換
318
319        Args:
320            value (`AcceptedTypes`): コンストラクタに与えられた引数
321
322        Raises:
323            TypeError: str型が変換できない場合
324
325        Returns:
326            datetime: 変換した型
327        """
328
329        if isinstance(value, ExtendedDatetime):
330            return value.dt
331        elif isinstance(value, datetime):
332            return value
333        elif isinstance(value, float):
334            return datetime.fromtimestamp(value)
335        elif isinstance(value, str):
336            try:
337                return datetime.fromisoformat(value)
338            except ValueError:
339                return datetime.strptime(value, "%Y/%m/%d %H:%M")
340        else:
341            raise TypeError("Unsupported type for datetime conversion")

345    """ExtendedDatetimeを要素とする日付リストを扱う補助クラス"""
346    @property
347    def start(self) -> ExtendedDatetime | None:
348        """リスト内の最小日付を返す。空ならNone。"""
349        return (min(self) if self else None)
350
351    @property
352    def end(self) -> ExtendedDatetime | None:
353        """リスト内の最大日付を返す。空ならNone。"""
354        return (max(self) if self else None)
355
356    def format(self, fmt: FormatType = "sql", delimiter: DelimiterStyle = None) -> list[str]:
357        """リストの全要素にformatを適用した文字列リストを返す"""
358        return [dt.format(fmt, delimiter) for dt in self if isinstance(dt, ExtendedDatetime)]
359
360    def dict_format(self, fmt: FormatType = "sql", delimiter: DelimiterStyle = None) -> dict[str, str]:
361        """リストの全要素にformatを適用した辞書を返す"""
362        date_range = [dt.format(fmt, delimiter) for dt in self if isinstance(dt, ExtendedDatetime)]
363
364        if not date_range:
365            return ({})
366
367        s = min(date_range)
368        e = max(date_range)
369
370        return {"start": s, "end": e}
