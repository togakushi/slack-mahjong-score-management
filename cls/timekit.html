cls.timekit

timekit - datetime 拡張ユーティリティ

  • ExtendedDatetime: 柔軟な初期化と書式変換ができる datetime 拡張クラス
  • ExtendedDatetimeList: ExtendedDatetimeを要素とする日付リストを扱う補助クラス
Examples:
>>> from cls.timekit import ExtendedDatetime
>>> t = ExtendedDatetime("2025-04-19 12:34:56")

>>> t.set("2025-05-01 00:00:00")
>>> t.format("sql")
'2025-05-01 00:00:00.000000'

>>> from dateutil.relativedelta import relativedelta
>>> t2 = t + relativedelta(days=93)

'2025/08/02'

>>> t + {"days": 1, "months": 2}
2025-07-02 00:00:00.000000

175@total_ordering
176class ExtendedDatetime:
177    """datetime拡張クラス"""
178
179    _dt: datetime
180    """操作対象"""
181
182    # 型アノテーション用定数
183    AcceptedType: TypeAlias = Union[str, float, datetime, "ExtendedDatetime"]
184    """引数として受け付ける型
185    - **str**: 日付文字列（ISO形式など）
186    - **float**: UNIXタイムスタンプ
187    - **datetime** / **ExtendedDatetime**: オブジェクトをそのまま利用
188    """
189
190    FormatType: TypeAlias = FormatType
191    DelimiterStyle: TypeAlias = DelimiterStyle
192
193    def __init__(self, value: Optional[AcceptedType] = None, **relativedelta_kwargs):
194        """ExtendedDatetimeの初期化
195
196        Args:
197            value (Optional[AcceptedType], optional): 引数. Defaults to None.
198                - None: 現在時刻(`datetime.now()`)で初期化
199            relativedelta_kwargs (dict): 初期化時にrelativedelta()に渡す引数
200        """
201
202        self._dt = self.convert(value) if value else datetime.now()
203        if relativedelta_kwargs:
204            self._dt += relativedelta(**relativedelta_kwargs)
205
206    def __str__(self) -> str:
207        return self.format("sql")
208
209    def __repr__(self) -> str:
210        return self.format("sql")
211
212    def __eq__(self, other):
213        if isinstance(other, ExtendedDatetime):
214            return self.dt == other.dt
215        if isinstance(other, datetime):
216            return self.dt == other
217        return NotImplemented
218
219    def __lt__(self, other):
220        if isinstance(other, ExtendedDatetime):
221            return self.dt < other.dt
222        if isinstance(other, datetime):
223            return self.dt < other
224        return NotImplemented
225
226    def __add__(self, other: Union[relativedelta, dict]) -> "ExtendedDatetime":
227        if isinstance(other, dict):
228            delta = relativedelta(**other)
229        elif isinstance(other, relativedelta):
230            delta = other
231        else:
232            raise TypeError("Expected dict or relativedelta")
233
234        return ExtendedDatetime(self._dt + delta)
235
236    def __sub__(self, other: Union[relativedelta, dict]) -> "ExtendedDatetime":
237        if isinstance(other, dict):
238            delta = relativedelta(**other)
239        elif isinstance(other, relativedelta):
240            delta = other
241        else:
242            raise TypeError("Expected dict or relativedelta")
243
244        return ExtendedDatetime(self._dt - delta)
245
246    def __radd__(self, other: Union[relativedelta, dict]) -> "ExtendedDatetime":
247        return self.__add__(other)
248
249    def __rsub__(self, other: Union[relativedelta, dict]) -> "ExtendedDatetime":
250        return self.__sub__(other)
251
252    def __hash__(self):
253        return hash(self.dt)
254
255    def __getattr__(self, name):
256        return getattr(self._dt, name)

277    def format(self, fmt: FormatType, delimiter: DelimiterStyle = None) -> str:
278        """フォーマット変換
279
280        Args:
281            fmt (FormatType): 変換形式
282            delimiter (DelimiterStyle): 区切り
283
284        Raises:
285            ValueError: 受け付けない変換形式
286
287        Returns:
288            str: 変換文字列
289        """
290
291        ret: str
292        match fmt:
293            case "ts":
294                ret = str(self._dt.timestamp())
295            case "y":
296                match delimiter:
297                    case "ja":
298                        ret = self._dt.strftime("%Y年")
299                    case _:
300                        ret = self._dt.strftime("%Y")
301            case "ym":
302                match delimiter:
303                    case "slash" | "/":
304                        ret = self._dt.strftime("%Y/%m")
305                    case "hyphen" | "-":
306                        ret = self._dt.strftime("%Y-%m")
307                    case "ja":
308                        ret = self._dt.strftime("%Y年%m月")
309                    case "number" | "num":
310                        ret = self._dt.strftime("%Y%m")
311                    case _:
312                        ret = self._dt.strftime("%Y/%m")
313            case "ymd":
314                match delimiter:
315                    case "slash" | "/":
316                        ret = self._dt.strftime("%Y/%m/%d")
317                    case "hyphen" | "-":
318                        ret = self._dt.strftime("%Y-%m-%d")
319                    case "ja":
320                        ret = self._dt.strftime("%Y年%m月%d日")
321                    case "number" | "num":
322                        ret = self._dt.strftime("%Y%m%d")
323                    case _:
324                        ret = self._dt.strftime("%Y/%m/%d")
325            case "ymdhm":
326                match delimiter:
327                    case "slash" | "/":
328                        ret = self._dt.strftime("%Y/%m/%d %H:%M")
329                    case "hyphen" | "-":
330                        ret = self._dt.strftime("%Y-%m-%d %H:%M")
331                    case "ja":
332                        ret = self._dt.strftime("%Y年%m月%d日 %H時%M分")
333                    case "number" | "num":
334                        ret = self._dt.strftime("%Y%m%d%H%M")
335                    case _:
336                        ret = self._dt.strftime("%Y/%m/%d %H:%M")
337            case "ymdhms":
338                match delimiter:
339                    case "slash" | "/":
340                        ret = self._dt.strftime("%Y/%m/%d %H:%M:%S")
341                    case "hyphen" | "-":
342                        ret = self._dt.strftime("%Y-%m-%d %H:%M:%S")
343                    case "ja":
344                        ret = self._dt.strftime("%Y年%m月%d日 %H時%M分%S秒")
345                    case "number" | "num":
346                        ret = self._dt.strftime("%Y%m%d%H%M%S")
347                    case _:
348                        ret = self._dt.strftime("%Y/%m/%d %H:%M:%S")
349            case "sql":
350                match delimiter:
351                    case "slash" | "/":
352                        ret = self._dt.strftime("%Y/%m/%d %H:%M:%S.%f")
353                    case "hyphen" | "-":
354                        ret = self._dt.strftime("%Y-%m-%d %H:%M:%S.%f")
355                    case "number" | "num":
356                        ret = self._dt.strftime("%Y%m%d%H%M%S%f")
357                    case _:
358                        ret = self._dt.strftime("%Y-%m-%d %H:%M:%S.%f")
359            case "ext":
360                ret = self._dt.strftime("%Y%m%d-%H%M%S")
361            case _:
362                raise ValueError(f"Unknown format: {fmt}")
363
364        return ret

366    def range(self, value: str | list) -> "ExtendedDatetimeList":
367        """キーワードが示す範囲をリストで返す
368
369        Args:
370            value (str | list): 範囲取得キーワード
371                - str: スペース区切りで分割してリスト化
372                - list: スペース区切りで再分割
373
374        Returns:
375            ExtendedDatetimeList: 日付リスト
376        """
377
378        if isinstance(value, str):
379            check_list = value.split()
380        else:
381            check_list = sum([str(x).split() for x in value], [])  # 平坦化
382
383        ret: list[datetime] = []
384        for word in check_list:
385            for _, range_map in DATE_RANGE_MAP.items():
386                if word in cast(list, range_map["keyword"]):
387                    ret.extend(range_map["range"](self._dt))
388                    break
389            else:
390                try:
391                    try_time = self.convert(str(word))
392                    ret.append(try_time.replace(hour=0, minute=0, second=0, microsecond=0))
393                    ret.append(try_time.replace(hour=23, minute=59, second=59, microsecond=999999))
394                except ValueError:
395                    pass
396
397            continue
398
399        return ExtendedDatetimeList([ExtendedDatetime(x) for x in ret])

415    @classmethod
416    def print_range(cls) -> str:
417        """指定可能キーワードで取得できる範囲の一覧
418
419        Returns:
420            str: 出力メッセージ
421        """
422
423        base_instance = cls()
424        ret: str = ""
425
426        for _, val in DATE_RANGE_MAP.items():
427            for label in val["keyword"]:
428                scope = " ～ ".join(base_instance.range(label).format("ymd"))
429                ret += f"{label}：{scope}\n"
430
431        return ret.strip()

433    @staticmethod
434    def convert(value: AcceptedType) -> datetime:
435        """引数の型を判定してdatetimeへ変換
436
437        Args:
438            value (AcceptedType): 変換対象
439
440        Raises:
441            TypeError: str型が変換できない場合
442
443        Returns:
444            datetime: 変換した型
445        """
446
447        if isinstance(value, ExtendedDatetime):
448            return value.dt
449        if isinstance(value, datetime):
450            return value
451        if isinstance(value, float):
452            return datetime.fromtimestamp(value)
453        if isinstance(value, str):
454            try:
455                return datetime.fromisoformat(value)
456            except ValueError:
457                return datetime.strptime(value, "%Y/%m/%d %H:%M")
458
459        raise TypeError("Unsupported type for datetime conversion")

462class ExtendedDatetimeList(list):
463    """ExtendedDatetimeを要素とする日付リストを扱う補助クラス"""
464
465    FormatType: TypeAlias = FormatType
466    DelimiterStyle: TypeAlias = DelimiterStyle
467
468    def __add__(self, other):
469        if isinstance(other, dict):
470            return ExtendedDatetimeList([dt + other for dt in self])
471        return NotImplemented
472
473    def __sub__(self, other):
474        if isinstance(other, dict):
475            return ExtendedDatetimeList([dt - other for dt in self])
476        return NotImplemented
477
478    @property
479    def start(self) -> ExtendedDatetime | None:
480        """最小日付を返す。空ならNone。"""
481        return min(self) if self else None
482
483    @property
484    def end(self) -> ExtendedDatetime | None:
485        """最大日付を返す。空ならNone。"""
486        return max(self) if self else None
487
488    @property
489    def period(self) -> List[ExtendedDatetime | None]:
490        """最小値と最大値をリストで返す"""
491        min_dt = min(self) if self else None
492        max_dt = max(self) if self else None
493
494        return [min_dt, max_dt]
495
496    def format(self, fmt: FormatType = "sql", delimiter: DelimiterStyle = None) -> list[str]:
497        """全要素にformatを適用した文字列リストを返す
498
499        Args:
500            fmt (FormatType, optional): フォーマット変換. Defaults to "sql".
501            delimiter (DelimiterStyle, optional): 区切り記号指定. Defaults to None.
502
503        Returns:
504            list[str]: 生成したリスト
505        """
506
507        return [dt.format(fmt, delimiter) for dt in self if isinstance(dt, ExtendedDatetime)]
508
509    def dict_format(self, fmt: FormatType = "sql", delimiter: DelimiterStyle = None) -> dict[str, str]:
510        """全要素にformatを適用し、最小日付と最大日付を辞書で返す
511
512        Args:
513            fmt (FormatType, optional): フォーマット変換. Defaults to "sql".
514            delimiter (DelimiterStyle, optional): 区切り記号指定. Defaults to None.
515
516        Returns:
517            dict[str, str]: 生成した辞書
518        """
519
520        date_range = [dt for dt in self if isinstance(dt, ExtendedDatetime)]
521
522        if not date_range:
523            return {}
524
525        return {"start": min(date_range).format(fmt, delimiter), "end": max(date_range).format(fmt, delimiter)}
